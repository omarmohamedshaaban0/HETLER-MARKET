// Search Auto-Suggest
const searchInput = document.getElementById('search');
const suggestions = document.getElementById('suggestions');
const mockSuggestions = ['Headphones', 'Shoes', 'Laptop'];
searchInput.addEventListener('input', () => {
    const query = searchInput.value.toLowerCase();
    suggestions.innerHTML = mockSuggestions.filter(s => s.toLowerCase().includes(query)).map(s => `<div>${s}</div>`).join('');
    suggestions.style.display = query ? 'block' : 'none';
});

// Currency/Language Switcher
document.getElementById('currency').addEventListener('change', (e) => localStorage.setItem('currency', e.target.value));
document.getElementById('language').addEventListener('change', (e) => localStorage.setItem('language', e.target.value));

// Live Chat
